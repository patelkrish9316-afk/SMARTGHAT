main_controller.c — ESP32 Master Loop 
/*
 * SmartGhat Road Safety System v2.0
 * File: main_controller.c
 * MCU: ESP32 (Xtensa LX6, 240MHz, dual-core)
 * Convergence 2026 — HT202603
 * ============================================
 * Main firmware entry point and control loop.
 * Handles sensor polling, decision logic,
 * LoRa coordination, and alert triggering.
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gpio.h"
#include "driver/ledc.h"
#include "esp_log.h"
#include "sensor_drivers.h"
#include "lora_comm.h"
#include "alert_engine.h"

/* ---- GPIO Pin Definitions ---- */
#define IR_SENSOR_PIN       GPIO_NUM_34   // IR proximity sensor input
#define ULTRASONIC_TRIG     GPIO_NUM_25   // HC-SR04 trigger
#define ULTRASONIC_ECHO     GPIO_NUM_26   // HC-SR04 echo
#define LED_WARNING_PIN     GPIO_NUM_2    // High-brightness LED array
#define LED_GREEN_PIN       GPIO_NUM_4    // Status green LED
#define BUZZER_PIN          GPIO_NUM_5    // Piezo buzzer output
#define SIGN_BOARD_CLK      GPIO_NUM_18   // LED matrix clock
#define SIGN_BOARD_DATA     GPIO_NUM_19   // LED matrix data

/* ---- Constants ---- */
#define DETECTION_RANGE_CM  500           // 5 meter detection zone
#define ALERT_DURATION_MS   10000         // Alert active for 10 seconds
#define POLL_INTERVAL_MS    50            // Sensor poll every 50ms
#define NODE_ID             'A'           // This node identifier (A or B)

/* ---- System State ---- */
typedef enum {
    STATE_STANDBY = 0,
    STATE_VEHICLE_DETECTED_LOCAL,
    STATE_VEHICLE_DETECTED_REMOTE,
    STATE_DUAL_DETECTION,               // CRITICAL: Both sides active
    STATE_ALERT_ACTIVE
} system_state_t;

static system_state_t current_state = STATE_STANDBY;
static bool remote_vehicle_present = false;
static bool local_vehicle_present  = false;
static uint32_t last_detection_time = 0;

static const char *TAG = "SMARTGHAT";

/* ---- Function Prototypes ---- */
void sensor_polling_task(void *pvParam);
void communication_task(void *pvParam);
void alert_management_task(void *pvParam);
void update_system_state(void);
void gpio_init_all(void);

/* ============================================================
 * ENTRY POINT
 * ============================================================ */
void app_main(void) {
    ESP_LOGI(TAG, "SmartGhat System v2.0 Booting...");
    ESP_LOGI(TAG, "Node ID: %c | MCU: ESP32 240MHz", NODE_ID);

    gpio_init_all();
    sensor_init();
    lora_init();
    alert_engine_init();

    /* Boot success indicator */
    gpio_set_level(LED_GREEN_PIN, 1);
    ESP_LOGI(TAG, "All subsystems initialized. Entering main loop.");

    /* Create FreeRTOS tasks on separate cores */
    xTaskCreatePinnedToCore(sensor_polling_task,  "SENSOR",  4096, NULL, 5, NULL, 0);
    xTaskCreatePinnedToCore(communication_task,   "LORA",    4096, NULL, 4, NULL, 0);
    xTaskCreatePinnedToCore(alert_management_task,"ALERT",   2048, NULL, 3, NULL, 1);
}

/* ============================================================
 * SENSOR POLLING TASK (Core 0)
 * Runs every 50ms — reads IR + ultrasonic sensors
 * ============================================================ */
void sensor_polling_task(void *pvParam) {
    sensor_data_t data;
    while (1) {
        read_sensors(&data);

        /* Vehicle detected if within detection range */
        bool detected = (data.ir_triggered ||
                         data.ultrasonic_cm < DETECTION_RANGE_CM);

        if (detected != local_vehicle_present) {
            local_vehicle_present = detected;
            last_detection_time = xTaskGetTickCount();

            if (detected) {
                ESP_LOGI(TAG, "LOCAL VEHICLE DETECTED — dist: %.1f cm",
                         data.ultrasonic_cm);
                /* Immediately broadcast to other node via LoRa */
                lora_broadcast_detection(NODE_ID, true);
            } else {
                ESP_LOGI(TAG, "Local vehicle cleared from zone");
                lora_broadcast_detection(NODE_ID, false);
            }
            update_system_state();
        }
        vTaskDelay(pdMS_TO_TICKS(POLL_INTERVAL_MS));
    }
}

/* ============================================================
 * COMMUNICATION TASK (Core 0)
 * Receives LoRa packets from the opposite node
 * ============================================================ */
void communication_task(void *pvParam) {
    lora_packet_t pkt;
    while (1) {
        if (lora_receive(&pkt)) {
            if (pkt.node_id != NODE_ID) {       // Message from other side
                remote_vehicle_present = pkt.vehicle_detected;
                ESP_LOGI(TAG, "REMOTE NODE %c: vehicle=%s",
                         pkt.node_id,
                         pkt.vehicle_detected ? "PRESENT" : "CLEAR");
                update_system_state();
            }
        }
        vTaskDelay(pdMS_TO_TICKS(10));
    }
}

/* ============================================================
 * STATE MACHINE UPDATE
 * Determines system alert level based on both nodes
 * ============================================================ */
void update_system_state(void) {
    system_state_t new_state;

    if (local_vehicle_present && remote_vehicle_present) {
        new_state = STATE_DUAL_DETECTION;       // ← CRITICAL COLLISION RISK
    } else if (local_vehicle_present) {
        new_state = STATE_VEHICLE_DETECTED_LOCAL;
    } else if (remote_vehicle_present) {
        new_state = STATE_VEHICLE_DETECTED_REMOTE;
    } else {
        new_state = STATE_STANDBY;
    }

    if (new_state != current_state) {
        current_state = new_state;
        ESP_LOGW(TAG, "State → %d", current_state);
        trigger_alert(current_state);           // Alert engine handles outputs
    }
}

/* ============================================================
 * GPIO INITIALIZATION
 * ============================================================ */
void gpio_init_all(void) {
    gpio_config_t io_conf = {};

    /* Outputs */
    io_conf.intr_type    = GPIO_INTR_DISABLE;
    io_conf.mode         = GPIO_MODE_OUTPUT;
    io_conf.pin_bit_mask = (1ULL << LED_WARNING_PIN) |
                           (1ULL << LED_GREEN_PIN)   |
                           (1ULL << BUZZER_PIN)      |
                           (1ULL << ULTRASONIC_TRIG);
    gpio_config(&io_conf);

    /* Inputs */
    io_conf.mode         = GPIO_MODE_INPUT;
    io_conf.pull_up_en   = 1;
    io_conf.pin_bit_mask = (1ULL << IR_SENSOR_PIN) |
                           (1ULL << ULTRASONIC_ECHO);
    gpio_config(&io_conf);

    ESP_LOGI(TAG, "GPIO configured successfully");
}
lora_comm.cpp — LoRa Mesh Communication 
/*
 * SmartGhat Road Safety System v2.0
 * File: lora_comm.cpp
 * Module: LoRa SX1276 Communication Layer
 * =========================================
 * Handles bi-directional LoRa communication
 * between Node A and Node B across the blind
 * curve. Range: up to 15km LOS in ghat terrain.
 * Frequency: 433MHz (India licensed band)
 */

#include "lora_comm.h"
#include <SPI.h>
#include <LoRa.h>
#include <ArduinoJson.h>

/* LoRa Hardware Pins (SPI) */
#define LORA_SS_PIN     18
#define LORA_RST_PIN    14
#define LORA_DIO0_PIN   26
#define LORA_FREQ       433E6       // 433 MHz

/* Protocol constants */
#define LORA_BANDWIDTH  125000      // 125kHz bandwidth
#define LORA_SPREAD     9           // Spreading factor 9
#define LORA_TXPOWER    20          // Max TX power (dBm)
#define MAGIC_BYTE      0xSG        // SmartGhat packet identifier
#define RETRY_COUNT     3           // Auto-retry on failure

static bool lora_ready = false;
static uint32_t tx_count = 0;
static uint32_t rx_count = 0;

/* ============================================================
 * INITIALIZATION
 * ============================================================ */
bool lora_init() {
    Serial.println("[LoRa] Initializing SX1276...");

    LoRa.setPins(LORA_SS_PIN, LORA_RST_PIN, LORA_DIO0_PIN);

    if (!LoRa.begin(LORA_FREQ)) {
        Serial.println("[LoRa] ERROR: Init failed! Check wiring.");
        return false;
    }

    /* Configure for maximum reliability in mountainous terrain */
    LoRa.setSpreadingFactor(LORA_SPREAD);
    LoRa.setSignalBandwidth(LORA_BANDWIDTH);
    LoRa.setCodingRate4(5);
    LoRa.setTxPower(LORA_TXPOWER);
    LoRa.enableCrc();               // Enable CRC for data integrity

    lora_ready = true;
    Serial.printf("[LoRa] Ready — Freq: 433MHz, SF:%d, BW:125kHz\n",
                  LORA_SPREAD);
    return true;
}

/* ============================================================
 * BROADCAST VEHICLE DETECTION TO OPPOSITE NODE
 * Sends a compact JSON packet via LoRa
 * ============================================================ */
bool lora_broadcast_detection(char node_id, bool vehicle_detected) {
    if (!lora_ready) return false;

    StaticJsonDocument<128> doc;
    doc["id"]  = node_id;
    doc["veh"] = vehicle_detected ? 1 : 0;
    doc["ts"]  = millis();
    doc["seq"] = ++tx_count;

    char buffer[128];
    serializeJson(doc, buffer, sizeof(buffer));

    bool success = false;
    for (int attempt = 0; attempt < RETRY_COUNT; attempt++) {
        LoRa.beginPacket();
        LoRa.write(0xAB);           // SmartGhat magic header
        LoRa.print(buffer);
        if (LoRa.endPacket(false)) {
            success = true;
            Serial.printf("[LoRa] TX[%lu] Node%c veh=%d RSSI OK\n",
                          tx_count, node_id, vehicle_detected);
            break;
        }
        delay(10);
    }

    if (!success) {
        Serial.println("[LoRa] WARNING: Transmission failed after retries");
    }
    return success;
}

/* ============================================================
 * RECEIVE PACKET FROM OPPOSITE NODE
 * Non-blocking — returns true if valid packet received
 * ============================================================ */
bool lora_receive(lora_packet_t *out_packet) {
    int packetSize = LoRa.parsePacket();
    if (packetSize == 0) return false;

    /* Read magic byte */
    if (LoRa.read() != 0xAB) {
        Serial.println("[LoRa] Invalid packet header, discarding");
        return false;
    }

    /* Read JSON payload */
    String payload = "";
    while (LoRa.available()) {
        payload += (char)LoRa.read();
    }

    StaticJsonDocument<128> doc;
    DeserializationError err = deserializeJson(doc, payload);
    if (err) {
        Serial.printf("[LoRa] JSON parse error: %s\n", err.c_str());
        return false;
    }

    out_packet->node_id          = doc["id"].as<char>();
    out_packet->vehicle_detected = doc["veh"].as<int>() == 1;
    out_packet->timestamp        = doc["ts"].as<uint32_t>();
    out_packet->rssi             = LoRa.packetRssi();
    out_packet->snr              = LoRa.packetSnr();

    rx_count++;
    Serial.printf("[LoRa] RX[%lu] NodeID:%c veh=%d RSSI:%d SNR:%.1f\n",
                  rx_count,
                  out_packet->node_id,
                  out_packet->vehicle_detected,
                  out_packet->rssi,
                  out_packet->snr);

    return true;
}

/* ============================================================
 * SIGNAL QUALITY DIAGNOSTICS
 * ============================================================ */
void lora_print_stats() {
    Serial.printf("[LoRa] Stats — TX: %lu | RX: %lu | RSSI: %d dBm\n",
                  tx_count, rx_count, LoRa.packetRssi());
}

/* ============================================================
 * TRIGGER ALERT — Called by state machine
 * Levels: 0=STANDBY, 1=ONE_SIDE, 2=DUAL_CRITICAL
 * ============================================================ */
void trigger_alert(system_state_t state) {
    switch (state) {
        case STATE_STANDBY:
            current_alert_level = 0;
            buzzer_off();
            set_sign_board("SAFE");
            led_set_duty(LEDC_CHANNEL_LED, 200);    // Dim green heartbeat
            Serial.println("[ALERT] System standby — All clear");
            break;

        case STATE_VEHICLE_DETECTED_LOCAL:
        case STATE_VEHICLE_DETECTED_REMOTE:
            current_alert_level = 1;
            buzzer_beep(2500, 300);                 // Single warning beep
            set_sign_board("CAUTION");
            led_set_duty(LEDC_CHANNEL_LED, 4096);   // Medium brightness
            Serial.println("[ALERT] CAUTION — One side vehicle detected");
            break;

        case STATE_DUAL_DETECTION:                  // ← CRITICAL STATE
            current_alert_level = 2;
            buzzer_pattern_critical();              // Rapid alarm pattern
            set_sign_board("STOP! VEH AHEAD");
            led_set_duty(LEDC_CHANNEL_LED, 8191);   // Maximum brightness
            Serial.println("[ALERT] !!! CRITICAL — DUAL VEHICLE DETECTED !!!");
            log_incident();                         // Write to SD card
            break;

        default:
            break;
    }
}

sensor_drivers.c — Multi-Sensor Detection 
/*
 * SmartGhat Road Safety System v2.0
 * File: sensor_drivers.c
 * Module: Multi-Sensor Detection Layer
 * ======================================
 * Drivers for:
 *   1. HC-SR04 Ultrasonic Distance Sensor
 *   2. HC-SR501 PIR Infrared Sensor
 *   3. Hall Effect Magnetic Sensor
 *   4. LDR Light Level Sensor (weather detection)
 * Sensor fusion logic for reliable detection
 * even in fog, dust, and rain conditions.
 */

#include "sensor_drivers.h"
#include "driver/gpio.h"
#include "esp_timer.h"
#include <math.h>

/* ---- Pin Definitions ---- */
#define ULTRASONIC_TRIG     GPIO_NUM_25
#define ULTRASONIC_ECHO     GPIO_NUM_26
#define IR_SENSOR_PIN       GPIO_NUM_34
#define HALL_SENSOR_PIN     GPIO_NUM_35
#define LDR_ADC_CHANNEL     ADC1_CHANNEL_6  // GPIO34

/* ---- Sensor Configuration ---- */
#define SPEED_OF_SOUND      34300   // cm/s at 20°C
#define ECHO_TIMEOUT_US     30000   // 30ms = ~5m max range
#define ULTRASONIC_SAMPLES  5       // Average 5 readings
#define TEMP_CORRECTION     0.0606  // cm/s per degree Celsius

/* Moving average buffer for noise reduction */
static float distance_buffer[ULTRASONIC_SAMPLES];
static uint8_t buf_index = 0;

/* ============================================================
 * SENSOR INITIALIZATION
 * ============================================================ */
void sensor_init(void) {
    /* Ultrasonic trig — output, echo — input */
    gpio_set_direction(ULTRASONIC_TRIG, GPIO_MODE_OUTPUT);
    gpio_set_direction(ULTRASONIC_ECHO, GPIO_MODE_INPUT);
    gpio_set_level(ULTRASONIC_TRIG, 0);

    /* IR and Hall sensors — input with pull-up */
    gpio_set_direction(IR_SENSOR_PIN, GPIO_MODE_INPUT);
    gpio_set_pull_mode(IR_SENSOR_PIN, GPIO_PULLUP_ONLY);
    gpio_set_direction(HALL_SENSOR_PIN, GPIO_MODE_INPUT);
    gpio_set_pull_mode(HALL_SENSOR_PIN, GPIO_PULLUP_ONLY);

    /* Initialize distance buffer */
    for (int i = 0; i < ULTRASONIC_SAMPLES; i++) {
        distance_buffer[i] = 9999.0f;  // Default = no vehicle
    }

    printf("[SENSOR] Multi-sensor array initialized\n");
}

/* ============================================================
 * READ HC-SR04 ULTRASONIC SENSOR
 * Returns distance in centimeters
 * -1.0 on timeout (no echo received)
 * ============================================================ */
float ultrasonic_measure_cm(float temp_celsius) {
    /* Temperature-compensated speed of sound */
    float sound_speed = SPEED_OF_SOUND + (TEMP_CORRECTION * temp_celsius);

    /* Send 10μs trigger pulse */
    gpio_set_level(ULTRASONIC_TRIG, 0);
    esp_rom_delay_us(2);
    gpio_set_level(ULTRASONIC_TRIG, 1);
    esp_rom_delay_us(10);
    gpio_set_level(ULTRASONIC_TRIG, 0);

    /* Measure echo pulse width with timeout */
    int64_t start_us = esp_timer_get_time();
    int64_t timeout  = start_us + ECHO_TIMEOUT_US;

    /* Wait for echo HIGH */
    while (!gpio_get_level(ULTRASONIC_ECHO)) {
        if (esp_timer_get_time() > timeout) return -1.0f;
    }
    int64_t echo_start = esp_timer_get_time();

    /* Wait for echo LOW */
    while (gpio_get_level(ULTRASONIC_ECHO)) {
        if (esp_timer_get_time() > timeout) return -1.0f;
    }
    int64_t echo_end = esp_timer_get_time();

    /* Calculate distance (divide by 2 for one-way) */
    float duration_us = (float)(echo_end - echo_start);
    float distance_cm = (duration_us * sound_speed) / (2.0f * 1000000.0f);

    return distance_cm;
}

/* ============================================================
 * SENSOR FUSION — Reads all sensors and returns fused result
 * Uses voting logic: 2 of 3 sensors must agree for detection
 * ============================================================ */
void read_sensors(sensor_data_t *out) {
    /* 1. Ultrasonic — moving average */
    float raw_dist = ultrasonic_measure_cm(25.0f);  // Assume 25°C
    distance_buffer[buf_index % ULTRASONIC_SAMPLES] = raw_dist;
    buf_index++;

    float avg_dist = 0;
    for (int i = 0; i < ULTRASONIC_SAMPLES; i++) {
        avg_dist += distance_buffer[i];
    }
    out->ultrasonic_cm = avg_dist / ULTRASONIC_SAMPLES;

    /* 2. IR PIR sensor */
    out->ir_triggered = (gpio_get_level(IR_SENSOR_PIN) == 0); // Active LOW

    /* 3. Hall effect (metallic vehicle detection) */
    out->hall_triggered = (gpio_get_level(HALL_SENSOR_PIN) == 0);

    /* 4. Light level (for automatic night/fog mode) */
    out->light_level = adc1_get_raw(LDR_ADC_CHANNEL);
    out->is_dark = (out->light_level < 800);

    /* Sensor fusion decision: majority voting */
    int vote = 0;
    if (out->ultrasonic_cm < 500) vote++;   // Within 5 meters
    if (out->ir_triggered)         vote++;
    if (out->hall_triggered)       vote++;

    out->vehicle_present = (vote >= 2);      // 2-of-3 confirms vehicle

    /* Weather compensation for ultrasonic */
    if (out->is_dark) {
        /* Reduce threshold sensitivity in night — fewer false positives */
        out->vehicle_present = (vote == 3);
    }
}
